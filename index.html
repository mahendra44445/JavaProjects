Logistic System - Java GUI Application

A Complete Logistics Management System with Swing GUI

GUI Features

  • Input fields for Shipment ID, Start Location, Vehicle ID, Driver Name, Item Name, Quantity
  • Buttons: Create Shipment, Assign Vehicle, Track Shipment, Start Delivery
  • Real-time status display and output console
  • Interactive user interface for logistics management

Complete Java GUI Code

import javax.swing.*; import java.awt.*; import java.util.HashMap; import java.util.Map; public class LogisticGUI extends JFrame { private JTextField shipmentIdField, startLocationField, vehicleIdField, driverNameField, itemNameField, quantityField; private JTextArea outputArea; private Warehouse warehouse; private Map<String, Shipment> shipments; private Map<String, Vehicle> vehicles; public LogisticGUI() { setTitle("User Defined Logistics Management System"); setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); setSize(1000, 600); setLocationRelativeTo(null); warehouse = new Warehouse(); shipments = new HashMap<>(); vehicles = new HashMap<>(); JPanel mainPanel = new JPanel(new BorderLayout(10, 10)); mainPanel.setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 10)); // Input Panel JPanel inputPanel = createInputPanel(); mainPanel.add(inputPanel, BorderLayout.NORTH); // Button Panel JPanel buttonPanel = createButtonPanel(); mainPanel.add(buttonPanel, BorderLayout.CENTER); // Output Area outputArea = new JTextArea(15, 50); outputArea.setEditable(false); JScrollPane scrollPane = new JScrollPane(outputArea); mainPanel.add(scrollPane, BorderLayout.SOUTH); add(mainPanel); setVisible(true); } private JPanel createInputPanel() { JPanel panel = new JPanel(new GridLayout(6, 2, 10, 10)); panel.setBorder(BorderFactory.createTitledBorder("Logistics Management Form")); panel.add(new JLabel("Shipment ID:")); shipmentIdField = new JTextField(); panel.add(shipmentIdField); panel.add(new JLabel("Start Location:")); startLocationField = new JTextField(); panel.add(startLocationField); panel.add(new JLabel("Vehicle ID:")); vehicleIdField = new JTextField(); panel.add(vehicleIdField); panel.add(new JLabel("Driver Name:")); driverNameField = new JTextField(); panel.add(driverNameField); panel.add(new JLabel("Item Name:")); itemNameField = new JTextField(); panel.add(itemNameField); panel.add(new JLabel("Quantity:")); quantityField = new JTextField(); panel.add(quantityField); return panel; } private JPanel createButtonPanel() { JPanel panel = new JPanel(new GridLayout(1, 4, 10, 10)); panel.setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 10)); JButton createButton = new JButton("Create Shipment"); createButton.addActionListener(e -> createShipment()); panel.add(createButton); JButton assignButton = new JButton("Assign Vehicle"); assignButton.addActionListener(e -> assignVehicle()); panel.add(assignButton); JButton trackButton = new JButton("Track Shipment"); trackButton.addActionListener(e -> trackShipment()); panel.add(trackButton); JButton deliveryButton = new JButton("Start Delivery"); deliveryButton.addActionListener(e -> startDelivery()); panel.add(deliveryButton); return panel; } private void createShipment() { String shipmentId = shipmentIdField.getText(); String location = startLocationField.getText(); String itemName = itemNameField.getText(); int quantity = Integer.parseInt(quantityField.getText()); Shipment shipment = new Shipment(shipmentId, location); shipments.put(shipmentId, shipment); warehouse.addItem(itemName, quantity); output("Shipment Created Successfully!"); output("Warehouse Inventory:"); warehouse.displayInventory(); } private void assignVehicle() { String vehicleId = vehicleIdField.getText(); String driverName = driverNameField.getText(); Vehicle vehicle = new Vehicle(vehicleId, driverName); vehicle.assign(); vehicles.put(vehicleId, vehicle); output("Vehicle Assigned Successfully!"); } private void trackShipment() { String shipmentId = shipmentIdField.getText(); Shipment shipment = shipments.get(shipmentId); if (shipment != null) { output("Shipment ID: " + shipmentId); output("Status: " + shipment.status); output("Location: " + shipment.location); output("Delivery started..."); } else { output("Shipment not found!"); } } private void startDelivery() { String shipmentId = shipmentIdField.getText(); Shipment shipment = shipments.get(shipmentId); if (shipment != null) { shipment.updateStatus("Out for Delivery"); shipment.updateLocation("On Route"); output("Shipment Delivered Successfully!"); } } private void output(String message) { outputArea.append(message + "\n"); } public static void main(String[] args) { SwingUtilities.invokeLater(() -> new LogisticGUI()); } }

Core Classes (Shipment, Vehicle, Warehouse)

static class Shipment { String shipmentId, status, location; Shipment(String shipmentId, String location) { this.shipmentId = shipmentId; this.location = location; this.status = "In Transit"; } } static class Vehicle { String vehicleId, driverName; boolean available; Vehicle(String vehicleId, String driverName) { this.vehicleId = vehicleId; this.driverName = driverName; this.available = true; } } static class Warehouse { Map<String, Integer> inventory = new HashMap<>(); void addItem(String item, int qty) { inventory.put(item, inventory.getOrDefault(item, 0) + qty); } }

How to Use

  • 1. Enter Shipment ID (e.g., "1")
  • 2. Enter Start Location (e.g., "hyderabad")
  • 3. Enter Vehicle ID (e.g., "201")
  • 4. Enter Driver Name (e.g., "gurucharan")
  • 5. Enter Item Name (e.g., "laptop")
  • 6. Enter Quantity (e.g., "1")
  • 7. Click "Create Shipment" button
  • 8. Click "Assign Vehicle" button
  • 9. Click "Track Shipment" button to track
  • 10. Click "Start Delivery" button to complete

Expected Output

Shipment Created Successfully! Warehouse Inventory: laptop: 1 Vehicle Assigned Successfully! Shipment ID: 1 Status: In Transit Location: hyderabad Delivery started... Shipment Delivered Successfully! Shipment ID: 1, Status: Out for Delivery, Location: On Route
